class Chrome:
    def __init__(self):
        pass
    def close_developer_tools(self):
        if exists():
            click(Pattern().targetOffset(-35,0))

    def open_developer_tools(self, orientation="vertical"):
        type("i", KeyModifier.CMD + KeyModifier.ALT)
        wait()
        if orientation == "vertical" and exists(Pattern().exact()):
            click(Pattern().exact())
            wait()
        elif orientation == "horizontal" and exists(Pattern().exact()):
            click(Pattern().exact())
            wait()

    def active_incognito(self):
        if exists():
            click(Pattern().targetOffset(0,-2))
        elif exists():
            click()
        else:
            click()
            type("n", KeyModifier.CMD + KeyModifier.SHIFT)

    def change_url(self, text):
        type("l", KeyModifier.CMD)
        type(text + Key.ENTER)
        wait()
        wait()

chrome = Chrome()
chrome.active_incognito()
chrome.close_developer_tools()
