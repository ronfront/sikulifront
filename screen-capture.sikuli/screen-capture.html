import time

class Chrome:
    def __init__(self):
        pass
    def close_developer_tools(self):
        if exists():
            click(Pattern().targetOffset(-35,0))

    def open_developer_tools(self, orientation="vertical"):
        type("i", KeyModifier.CMD + KeyModifier.ALT)
        wait()
        if orientation == "vertical" and exists(Pattern().exact()):
            click(Pattern().exact())
            wait()
        elif orientation == "horizontal" and exists(Pattern().exact()):
            click(Pattern().exact())
            wait()

    def active_incognito(self):
        if exists(Pattern().exact()):
            click(Pattern().exact())
        elif exists():
            click()
        else:
            click()
            type("n", KeyModifier.CMD + KeyModifier.SHIFT)


    def active_regular(self):
        if exists(Pattern().targetOffset(-8,-3)):
            click(Pattern().targetOffset(-8,-3))
        elif exists(Pattern().targetOffset(-11,-12)):
            click(Pattern().targetOffset(-11,-12))

    def change_url(self, text):
        type("l", KeyModifier.CMD)
        type(text + Key.ENTER)
        self.wait_page_load()

    def wait_page_load(self):
        if exists():
            waitVanish()
        wait()
